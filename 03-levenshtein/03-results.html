def plotHistogram(values, min_val, max_val, bins, show_info=False, logarithmic=True):
    events, edges, patches = plt.hist(values, range=(min_val, max_val), bins=bins, log=logarithmic)
    if show_info:
        print("Elements matched:")
        print(events)
        print("Bins boundaries:")
        print(edges)
        print("Patches:")
        for p in patches:
            print(p)

    plt.xlabel('single word occurances in all judgements (range [%.2f, %.2f])' % (min_val, max_val), 
               fontsize=15, 
               labelpad=20)
    plt.ylabel('number of words', fontsize=15)
    plt.title('Words occurances in judgments (2015) histogram')
    plt.rcParams["figure.figsize"] = [14,7]
    plt.rcParams.update({'font.size': 17})
    plt.xticks(edges, rotation=90)
    plt.rcParams['xtick.labelsize'] = 12
    plt.rcParams['ytick.labelsize'] = 12
    plt.rcParams['figure.figsize'] = 20, 10
    plt.show()

import csv

def getIncorrectWords():
    freq_list_dir = "data/results2015.csv"
    with open(freq_list_dir, mode='r') as infile:
        reader = csv.reader(infile)
        freq_list = {rows[0]: rows[1] for rows in reader}
    incorrect_words = freq_list.keys() - correct_words
    return incorrect_words

import editdistance as ed

def findCorrectWords(words):
    for w1 in words:
        neighboring_words = []
        not_found = True
        thresh = 1
        while not_found:         
            neighboring_words = [w2 for w2 in correct_words if ed.eval(w1, w2) <= thresh]
            if not neighboring_words:
                thresh += 1
            else:
                not_found = False
        print("original: ", w1, "\tcorrect: ", neighboring_words, "\tdistance: ", thresh)
